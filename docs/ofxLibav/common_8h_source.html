218 static av_always_inline av_const int av_popcount_c(uint32_t x)
219 {
220  x -= (x >> 1) & 0x55555555;
221  x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
222  x = (x + (x >> 4)) & 0x0F0F0F0F;
